from Packages.flask import Flask, request
import socket
from Pypi import Installer
app = Flask(__name__)
 
@app.route('/')
def index():
	return 'Hello World!'
	
@app.route('/install')
def install():
	package = request.args.get('package', '')
	version = request.args.get('version', '')
	i = Installer(package, version)
	i.install()
	return 'Successfully installed ' + package
